# ediop3PHISH
A phishing tool in bash. It doesn't require root. Works on termux. 


# for termux

git clone https://github.com/ediop3SquadALT/ediop3PHISH

cd ediop3PHISH

pkg install zip

chmod +x ./termux.sh

./termux.sh

unzip ediop3PHISH.sh

cd ediop3PHISH

bash ediop3PHISH.sh



# for linux

git clone https://github.com/ediop3SquadALT/ediop3PHISH

cd ediop3PHISH

sudo apt install zip

chmod +x ./Linux.sh

./Linux.sh

unzip ediop3PHISH.sh

cd ediop3PHISH.sh

sudo bash ediop3PHISH.sh
